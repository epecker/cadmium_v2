121  if (cellModel == nullptr) {
122  throw CadmiumModelException("Scenario component is not a cell");
123  }
124  for (const auto& neighbor: cellModel->getNeighborhood()) {
125  addIC(cellId(neighbor.first), "outputNeighborhood", cellModel->getId(), "inputNeighborhood");
126  }
127  auto cellConfig = cellModel->getCellConfig();
128  for (const auto& [portFrom, portTo]: cellConfig->EIC) {
129  addDynamicEIC(portFrom, cellModel->getId(), portTo); // TODO
130  }
131  for (const auto& portTo: cellConfig->EOC) {
132  addDynamicEOC(cellModel->getId(), "outputNeighborhood", portTo); // TODO
133  }
134  }
