159  void addCoupling(const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
160  if (!portTo->compatible(portFrom)) {
161  throw CadmiumModelException("invalid port type");
162  }
163  if (portFrom->getParent() == nullptr || portTo->getParent() == nullptr) {
164  throw CadmiumModelException("port does not belong to any model");
165  }
166  auto portFromParent = portFrom->getParent();
167  auto portToParent = portTo->getParent();
168  if (containsInPort(portFrom)) {
169  if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
170  addCoupling(EIC, portFrom, portTo);
171  } else {
172  throw CadmiumModelException("invalid destination port");
173  }
174  } else if (portFromParent->getParent() == this && portFromParent->containsOutPort(portFrom)) {
175  if (containsOutPort(portTo)) {
176  addCoupling(EOC, portFrom, portTo);
177  } else if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
178  addCoupling(IC, portFrom, portTo);
179  } else {
180  throw CadmiumModelException("invalid destination port");
181  }
182  } else {
183  throw CadmiumModelException("invalid origin port");
184  }
185  }
